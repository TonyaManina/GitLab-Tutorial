int main() {
    double x;
    char some_symb = '\n';
    int valid_scan = scanf("%lg%c", &x, &some_symb);
    if (valid_scan == 2 && some_symb == '\n') {
        if (x) {
            printf("%.1f", important_func(x));
            return 0;
        }
    }

    printf("n/a");
    return 1;
}
double important_func(double x) {
    int a, b;
    a = 7 * pow(10, -3) * pow(x, 4) + ((22.8 * pow(x, 1 / 3) - pow(10, 3)) * x + 3) / (x * x / 2);
    b = x * pow((10 + x), 2 / x) - 1.01;
    return a - b;
}}
